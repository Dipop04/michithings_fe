git config core.autocrlf true: Cambiar la interpretación de caracter

git init: Inicializar el repositorio
git status: Revisar todo
git add (ARCHIVO): añadirlo
git add . : Añade todo
git checkout -- .: Reconstruir el anterior commit
git branch: ver qué ramas existen y en qué rama estoy trabajando actualmente

git config --global init.defaultBranch main: Cambiar la rama principal de master a main


git config --global alias.lg "log --graph --abbrev-commit --decorate --format=format:'%C(bold blue)%h%C(reset) - %C(bold green)(%ar)%C(reset) %C(white)%s%C(reset) %C(dim white)- %an%C(reset)%C(bold yellow)%d%C(reset)' --all"


git diff: Saber qué modificaciones he realizado
git commit -am "": añadir y commit
git commit -amend -m "": Corregir commit
git reset --soft HEAD^:Borrar el commit anterior para añadir nuevas modificaciones

RECOMENDACIÓN: Realizar modificaciones en los últimos commit

git reset --soft [Agregar id de commit anterior al que queremos modificar]
git reflog: todo lo que ha sucedido en orden cromológico

git mv [Nombre archivo] [Nombre nuevo]: mover o cambiar nombre de archivo
git rm [Nombre archivo]: Remover archivo
